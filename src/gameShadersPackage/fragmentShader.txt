#version 400 core
//This is the fragment shader which takes the color values of several vertexes and blends accordingly
//Input texture coordinates
in vec2 pass_placementTexture;
//Output Colored Pixels
out vec4 out_Color;

uniform sampler2D textureSampler;

void main(void){
	//Output the texture mapped to the proper coordinates
	out_Color = texture(textureSampler, pass_placementTexture);
}
